<Tabs>
  <Tab title="JavaScript/TypeScript">
  <Info>
    **Prerequisites**:
    - Node.js version 19 or higher
  </Info>



  </Tab>

  <Tab title="Python">
  <Info>
    **Prerequisites**:
    - Python version 3.8 or higher
  </Info>

  ```bash
  pip install fintag
  ```
  </Tab>

  <Tab title="Rust">
  <Info>
    **Prerequisites**:
    - Rust version 1.60 or higher
  </Info>

  ```bash
  cargo add fintag
  ```
  </Tab>

  <Tab title="Go">
  <Info>
    **Prerequisites**:
    - Go version 1.18 or higher
  </Info>

  ```bash
  go get github.com/fintag-io/fintag-go
  ```
  </Tab>

  <Tab title="Swift">
  <Info>
    **Prerequisites**:
    - Swift version 5.5 or higher
  </Info>

  ```bash
  swift package add @fintag-io/fintag-swift
  ```
  </Tab>

</Tabs>
